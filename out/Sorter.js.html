/**
 * Sort messages by date
 * @param {Array} list of Objects messages
 * @return {Array} ordered list 
 */
function sort(list) {
    var sorted = linq.from(list).orderBy(function (m) {
        return m.timestamp;
    }).toArray();
    return sorted;
}

exports.sort = sort;
